` Iterators 迭代器`
function breakfast(foods) {
  let i = 0
  return {
    next() {
      let done = i >= foods.length
      let value = done ? undefined : foods[i++]
      return {
        value: value,
        done: done
      }
    }
  }
}
let sunny = breakfast(['apple', 'bananer'])
console.log(sunny.next())
console.log(sunny.next())
console.log(sunny.next())

    this.name = name
    this.birthday = birthday
  }

  }

class Other extends Person {

    super(name, birthday)
  }
